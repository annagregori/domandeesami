    <script>

        // --- CONFIGURA I TUOI DATI QUI ---

        // Puoi aggiungere, modificare o rimuovere queste righe.

        const databaseData = [

            {

                testo: "Comprare il pane e latte",

                tag1: { testo: "Personale", colore: "bg-blue-100 text-blue-800" },

                tag2: { testo: "Da fare", colore: "bg-gray-100 text-gray-800" },

                data: "2026-05-25"

            },

            {

                testo: "Finire il progetto GitHub Pages",

                tag1: { testo: "Lavoro", colore: "bg-purple-100 text-purple-800" },

                tag2: { testo: "In Corso", colore: "bg-yellow-100 text-yellow-800" },

                data: "2026-06-01"

            },

            {

                testo: "Prenotare le vacanze estive",

                tag1: { testo: "Viaggi", colore: "bg-green-100 text-green-800" },

                tag2: { testo: "Pronto", colore: "bg-green-100 text-green-800" },

                data: "2026-07-15"

            }

        ];


        // --- FUNZIONE PER COSTRUIRE LA TABELLA ---

        function renderDatabase() {

            const tbody = document.getElementById('database-body');

            tbody.innerHTML = ''; // Svuota la tabella


            databaseData.forEach(row => {

                // Formattiamo la data in modo più leggibile (es. 25 mag 2026)

                const opzioniData = { year: 'numeric', month: 'short', day: 'numeric' };

                const dataFormattata = new Date(row.data).toLocaleDateString('it-IT', opzioniData);


                const tr = document.createElement('tr');

                tr.className = "hover:bg-[#f7f7f5] transition-colors duration-150";


                tr.innerHTML = `

                    <td class="px-4 py-3 border-r border-gray-200 font-medium text-gray-900">${row.testo}</td>

                    <td class="px-4 py-3 border-r border-gray-200">

                        <span class="px-2 py-1 rounded text-xs font-medium ${row.tag1.colore}">${row.tag1.testo}</span>

                    </td>

                    <td class="px-4 py-3 border-r border-gray-200">

                        <span class="px-2 py-1 rounded text-xs font-medium ${row.tag2.colore}">${row.tag2.testo}</span>

                    </td>

                    <td class="px-4 py-3 text-gray-500">${dataFormattata}</td>

                `;

                tbody.appendChild(tr);

            });

        }


        // Avvia il rendering quando la pagina si carica

        document.addEventListener('DOMContentLoaded', renderDatabase);
